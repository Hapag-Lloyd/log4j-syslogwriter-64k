3   import java.io.IOException;
4   import java.net.DatagramPacket;
5   import java.net.DatagramSocket;
6   import java.net.SocketException;
7   import java.nio.charset.Charset;
8   
9   import org.apache.log4j.helpers.LogLog;
10  
11  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
12  
13  /**
14   * SyslogWriter64k is a wrapper around the java.net.DatagramSocket class so that
15   * it behaves like a java.io.Writer.
16   */
17  @SuppressFBWarnings(value = "IMC_IMMATURE_CLASS_NO_TOSTRING",
18  		justification = "Instance fields cannot be stringified nicely, therefore toString does not make that much sense.")
19  public class SyslogUdpWriter64k extends SyslogWriter64k {
20  	private final DatagramSocket socket;
21  
22  	@SuppressWarnings({ "PMD.CloseResource", "PMD.GuardLogStatement" })
23  	public SyslogUdpWriter64k(final String syslogHost, final Charset charset) {
24  		super(syslogHost, charset);
25  
26  		DatagramSocket udpSocket = null;
27  		try {
28  			udpSocket = new DatagramSocket();
29  		} catch (final SocketException e) {
30  			LogLog.error("Could not instantiate DatagramSocket to " + syslogHost + ". All logging will FAIL.", e);
31  		}
32  		this.socket = udpSocket;
33  	}
34  
35  	@Override
36  	public void write(final char[] buf, final int off, final int len) throws IOException {
37  		this.write(new String(buf, off, len));
38  	}
39  
40  	@Override
41  	public void write(final String string) throws IOException {
42  		final byte[] bytes = string.getBytes(getCharset());
43  		final DatagramPacket packet = new DatagramPacket(bytes, bytes.length, getSyslogHost(), getSyslogPort());
44  
45  		if (this.socket != null) {
46  			socket.send(packet);
47  		}
48  	}
49  
50  	@Override
51  	public void flush() {
52  		// nothing to flush
53  	}
54  
55  	@Override
56  	@SuppressWarnings("PMD.CloseResource")
57  	public void close() {
58  		socket.close();
59  	}
60  }
