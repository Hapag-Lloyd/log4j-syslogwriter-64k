4   import java.io.IOException;
5   import java.io.OutputStreamWriter;
6   import java.net.Socket;
7   import java.nio.charset.Charset;
8   import java.time.Duration;
9   import java.util.Optional;
10  import java.util.concurrent.atomic.AtomicReference;
11  
12  import javax.net.SocketFactory;
13  
14  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
15  
16  /**
17   * SyslogWriter64k is a wrapper around the java.net.DatagramSocket class so that
18   * it behaves like a java.io.Writer.
19   */
20  @SuppressWarnings("PMD.CloseResource")
21  @SuppressFBWarnings(value = "IMC_IMMATURE_CLASS_NO_TOSTRING",
22  		justification = "Instance fields cannot be stringified nicely, therefore toString does not make that much sense.")
23  public class SyslogTcpWriter64k extends SyslogWriter64k {
24  	private final Optional<SocketFactory> socketFactory;
25  
26  	private final Duration socketTimeout;
27  
28  	private final AtomicReference<Socket> socket = new AtomicReference<>(null);
29  
30  	private final AtomicReference<BufferedWriter> writer = new AtomicReference<>(null);
31  
32  	public SyslogTcpWriter64k(final String syslogHost,
33  			final Charset charset,
34  			final Optional<SocketFactory> socketFactory,
35  			final Duration socketTimeout) {
36  		super(syslogHost, charset);
37  
38  		this.socketFactory = socketFactory;
39  		this.socketTimeout = socketTimeout;
40  	}
41  
42  	@Override
43  	@SuppressWarnings({ "checkstyle:SuppressWarnings", "PMD.CloseResource", "resource" })
44  	public void flush() throws IOException {
45  		final BufferedWriter writerToFlush = writer.get();
46  		if (writerToFlush != null) {
47  			closeOnIOException(writerToFlush::flush);
48  		}
49  	}
50  
51  	@SuppressWarnings({ "checkstyle:SuppressWarnings", "resource" })
52  	@SuppressFBWarnings(value = { "OI_OPTIONAL_ISSUES_USES_IMMEDIATE_EXECUTION", "UNENCRYPTED_SOCKET" },
53  			justification = "false-positive, as '0' is constant; Offering both: insecure TCP and secure TCP via custom SocketFactory")
54  	private BufferedWriter getWriter() throws IOException {
55  		synchronized (lock) {
56  			if (writer.get() == null) {
57  				final Socket socketToSet = socketFactory.isPresent()
58  						? socketFactory.get().createSocket(getSyslogHost(), getSyslogPort())
59  						: new Socket(getSyslogHost(), getSyslogPort());
60  				socketToSet.setSoTimeout((int) socketTimeout.toMillis());
61  				socket.set(socketToSet);
62  
63  				writer.set(new BufferedWriter(new OutputStreamWriter(socketToSet.getOutputStream(), getCharset())));
64  			}
65  			return writer.get();
66  		}
67  	}
68  
69  	@Override
70  	public void write(final String string) throws IOException {
71  		// compute syslog frame according to: https://tools.ietf.org/html/rfc6587
72  		final String syslogFrame = String.format("%s %s", string.length(), string);
73  		closeOnIOException(() -> getWriter().append(syslogFrame));
74  	}
75  
76  	@Override
77  	public void write(final char[] buf, final int off, final int len) throws IOException {
78  		this.write(new String(buf, off, len));
79  	}
80  
81  	@Override
82  	@SuppressWarnings("PMD.EmptyControlStatement")
83  	@SuppressFBWarnings(value = "AFBR_ABNORMAL_FINALLY_BLOCK_RETURN", justification = "Shouldn't matter in this case.")
84  	public void close() throws IOException {
85  		synchronized (lock) {
86  			// Using try-with-resource to avoid null checks and nested finally blocks
87  			try (@SuppressWarnings("PMD.UnusedLocalVariable")
88  			Socket socketToClose = socket.get();
89  					@SuppressWarnings("PMD.UnusedLocalVariable")
90  					BufferedWriter writerToClose = writer.get()) {
91  				// nothing
92  			} finally {
93  				writer.set(null);
94  				socket.set(null);
95  			}
96  		}
97  	}
98  
99  	private void closeOnIOException(final IORunnable runnable) throws IOException {
100 		try {
101 			runnable.run();
102 		} catch (final IOException e) {
103 			try {
104 				close();
105 			} catch (@SuppressWarnings("unused") final IOException ignored) {
106 				// ignore because it shall not hide the original exception
107 			}
108 			throw e;
109 		}
110 	}
111 
112 	@FunctionalInterface
113 	private interface IORunnable {
114 		void run() throws IOException;
115 	}
116 }
